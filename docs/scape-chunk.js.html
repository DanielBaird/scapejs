/**
 * Represents a rectangular prism of scape material -- dirt or
 * whatever.
 * @class
 */
function ScapeChunk(scene, layer, material, options) {

    var defaultOptions = {};
    // invoke our super constructor
    ScapeObject.call(this, options, defaultOptions);

    this._scene = scene;
    this._layer = layer;
    this._material = material;
    this._createObject();
    this._addObject();
};
